\subsubsection{Baselines}

Comparamos contra:

\begin{enumerate}
    \item \textbf{Linear Regression / Logistic}: Modelo tradicional sem destilacao
    \item \textbf{GAM Vanilla}: GAM treinado diretamente nos dados (sem destilacao)
    \item \textbf{Standard KD}: Knowledge distillation classica (sem restricoes economicas)
    \item \textbf{Teacher Model}: XGBoost de alta acuracia (limite superior)
\end{enumerate}

\subsubsection{Metricas}

\begin{itemize}
    \item \textbf{Acuracia Preditiva}: AUC-ROC, F1-score, KS statistic
    \item \textbf{Estabilidade}: CV de coeficientes, sign stability
    \item \textbf{Compliance Economica}: Taxa de conformidade com restricoes
    \item \textbf{Interpretabilidade}: Economic Interpretability Score (0-100\%)
\end{itemize}

\subsection{Case Study 1: Risco de Credito}

\subsubsection{Contexto}

\textbf{Problema}: Bancos precisam modelos de credit scoring que:
\begin{itemize}
    \item Alcancem acuracia competitiva (regulacao Basel III)
    \item Produzam coeficientes interpretaveis para reguladores
    \item Respeitem relacoes economicas (income $\uparrow$ $\rightarrow$ default $\downarrow$)
\end{itemize}

\textbf{Dataset}: 250,000 emprestimos (2005-2015), 42 features economicas, target = default binario.

\subsubsection{Restricoes Economicas Especificadas}

\begin{table}[h]
\centering
\caption{Restricoes Economicas - Credito}
\begin{tabular}{lll}
\toprule
\textbf{Feature} & \textbf{Restricao} & \textbf{Justificativa} \\
\midrule
Income & Sign: Negativo & Maior renda $\rightarrow$ menor risco \\
DTI Ratio & Sign: Positivo & Maior endividamento $\rightarrow$ maior risco \\
Interest Rate & Sign: Positivo & Taxa alta indica risco percebido \\
Age & Monotonia crescente (18-65) & Maturidade financeira \\
Employment Length & Monotonia crescente & Estabilidade profissional \\
\bottomrule
\end{tabular}
\end{table}

\subsubsection{Resultados - Acuracia Preditiva}

\begin{table}[h]
\centering
\caption{Resultados - Risco de Credito}
\begin{tabular}{lccc}
\toprule
\textbf{Modelo} & \textbf{AUC-ROC} & \textbf{F1} & \textbf{KS Stat} \\
\midrule
Logistic Regression & 0.782 & 0.654 & 0.421 \\
GAM Vanilla & 0.801 & 0.683 & 0.458 \\
Standard KD (GAM) & 0.836 & 0.721 & 0.512 \\
\textbf{Economic KD (GAM)} & \textbf{0.829} & \textbf{0.715} & \textbf{0.506} \\
Teacher (XGBoost) & 0.847 & 0.731 & 0.523 \\
\midrule
\multicolumn{4}{l}{\textit{Perda vs. Teacher: -2.1\% AUC, -2.2\% F1}} \\
\multicolumn{4}{l}{\textit{Ganho vs. GAM Vanilla: +3.5\% AUC, +4.7\% F1}} \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Observacao}: Economic KD alcanca 97.9\% da acuracia do teacher, superando GAM vanilla em 3.5\% AUC.

\begin{itemize}

    \begin{itemize}
        \item Pre-2008: $\beta_{\text{DTI}} = +0.412$
        \item Pos-2008: $\beta_{\text{DTI}} = +0.627$ (+52\% aumento)

    \item \textbf{Interpretacao Economica}: Crise aumentou sensibilidade de risco a endividamento
\end{itemize}

\subsection{Case Study 2: Economia do Trabalho}

\subsubsection{Contexto}

\textbf{Problema}: Analise de impacto de politicas de emprego (e.g., salario minimo) requer modelos com:
\begin{itemize}
    \item Efeitos marginais interpretaveis
    \item Conformidade com teoria de busca de emprego
    \item Capacidade de predicao para targeting de programas
\end{itemize}

\textbf{Dataset}: 180,000 individuos, 38 features socioeconomicas, target = empregado (binario).

\subsubsection{Resultados}

\begin{table}[h]
\centering
\caption{Resultados - Economia do Trabalho}
\begin{tabular}{lcccc}
\toprule
\textbf{Modelo} & \textbf{AUC} & \textbf{F1} & \textbf{Avg CV} & \textbf{Compliance} \\
\midrule
Logistic & 0.724 & 0.681 & --- & 82\% \\
GAM Vanilla & 0.751 & 0.702 & --- & 89\% \\
Standard KD & 0.788 & 0.741 & 0.203 & 76\% \\
\textbf{Economic KD} & \textbf{0.783} & \textbf{0.736} & \textbf{0.124} & \textbf{96\%} \\
Teacher (XGBoost) & 0.801 & 0.753 & --- & --- \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Insights}:
\begin{itemize}
    \item Economic KD: 97.8\% da acuracia do teacher
    \item Compliance economica: 96\% (vs. 76\% do KD padrao)
    \item Estabilidade superior: CV 0.124 vs. 0.203 (Standard KD)
\end{itemize}

\subsubsection{Efeitos Marginais - Educacao}

\begin{itemize}
    \item \textbf{High School}: +8.2\% probabilidade de emprego
    \item \textbf{Bachelor's}: +17.5\% (adicional sobre HS)
    \item \textbf{Master's+}: +24.1\% (adicional sobre HS)
    \item \textbf{Conformidade}: Monotonia crescente preservada em 100\% dos bootstrap samples
\end{itemize}

\subsection{Ablation Study}

\subsubsection{Impacto de Restricoes Economicas}

Removendo componentes do framework (dataset: Credito):

\begin{table}[h]
\centering
\caption{Ablation Study - Contribuicao de Componentes}
\begin{tabular}{lccc}
\toprule
\textbf{Configuracao} & \textbf{AUC} & \textbf{Compliance} & \textbf{Avg CV} \\
\midrule
Economic KD (Full) & 0.829 & 96\% & 0.116 \\
- Sign Constraints & 0.831 & 82\% & 0.121 \\
- Monotonicity Constraints & 0.830 & 87\% & 0.118 \\
- Constraint Loss Term & 0.834 & 74\% & 0.187 \\
Standard KD (No Economics) & 0.836 & 76\% & 0.203 \\
\bottomrule
\end{tabular}
\end{table}

\textbf{Insights}:
\begin{itemize}
    \item Restricoes economicas custam 0.7\% AUC, mas ganham +20pp compliance
    \item Constraint loss term e critico para estabilidade (CV 0.116 vs. 0.187)
\end{itemize}

\subsection{Reproducibilidade}

    \item \textbf{AUC}: $0.829 \pm 0.003$ (std muito baixo)
    \item \textbf{Compliance}: $96\% \pm 1.2\%$
    \item \textbf{Avg CV}: $0.116 \pm 0.008$
\end{itemize}

\textbf{Conclusao}: Resultados altamente reproduziveis.
